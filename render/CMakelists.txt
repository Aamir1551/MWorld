cmake_minimum_required(VERSION 3.10)
set(CMAKE_CXX_STANDARD 11)

project(render)
add_subdirectory("lib/glad")
add_subdirectory("${CMAKE_CURRENT_SOURCE_DIR}/../utils" "./build")

add_executable(tut main.cpp)
add_executable(ans ans.cpp)

target_include_directories(tut PUBLIC "${CMAKE_CURRENT_SOURCE_DIR}/lib/glfw-3.3.2/include/")
target_include_directories(ans PUBLIC "${CMAKE_CURRENT_SOURCE_DIR}/lib/glfw-3.3.2/include/")

target_include_directories(tut PUBLIC "${CMAKE_CURRENT_SOURCE_DIR}/lib/glad/include/")
target_include_directories(ans PUBLIC "${CMAKE_CURRENT_SOURCE_DIR}/lib/glad/include/")

target_link_libraries(ans "${CMAKE_CURRENT_SOURCE_DIR}/lib/glfw-3.3.2/build/src/libglfw3.a")
target_link_libraries(tut "${CMAKE_CURRENT_SOURCE_DIR}/lib/glfw-3.3.2/build/src/libglfw3.a")
target_link_libraries(tut utils)

target_link_libraries(ans glad)
target_link_libraries(tut glad)
